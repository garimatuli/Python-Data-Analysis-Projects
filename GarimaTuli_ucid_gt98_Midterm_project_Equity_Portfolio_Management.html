# Calculating remaining cash after buying maximum amount of shares for the 5 stocks

def calculate_stock_vals(stock,row,spend_on_each_stock):
    share_count = spend_on_each_stock // universe[stock][row]
    cost_of_buying_max_shares = share_count * universe[stock][row]
    cash_remain = spend_on_each_stock - cost_of_buying_max_shares
    
    return share_count,cost_of_buying_max_shares,cash_remain

def calculate_initial_MTM(i,stock1,stock2,stock3,stock4,stock5,strategy):
    global share_count_stock1
    global cost_of_buying_max_shares_stock1
    global cash_remain_stock1

    global share_count_stock2
    global cost_of_buying_max_shares_stock2
    global cash_remain_stock2

    global share_count_stock3
    global cost_of_buying_max_shares_stock3
    global cash_remain_stock3

    global share_count_stock4
    global cost_of_buying_max_shares_stock4
    global cash_remain_stock4

    global share_count_stock5
    global cost_of_buying_max_shares_stock5
    global cash_remain_stock5

    global cash
    global cost
    
    if i==0:
        spend_on_each_stock = 1000000
    else:
        if strategy == "low":
            spend_on_each_stock = (rebalancing_dataframe_low['MTM'][i-1]) / 5.0
        else:
            spend_on_each_stock = (rebalancing_dataframe_high['MTM'][i-1]) / 5.0
            
    # Calculating maximum number of shares which can be brought in 1 million for each of ['stock1','stock2','stock3','stock4','stock5']
    # Calculating total cost of buying maximum amount of shares for the 5 stocks
    # Calculating remaining cash after buying maximum amount of shares for the 5 stocks
    
    share_count_stock1, cost_of_buying_max_shares_stock1, cash_remain_stock1 = calculate_stock_vals(stock1,i,spend_on_each_stock)
    share_count_stock2, cost_of_buying_max_shares_stock2, cash_remain_stock2 = calculate_stock_vals(stock2,i,spend_on_each_stock)
    share_count_stock3, cost_of_buying_max_shares_stock3, cash_remain_stock3 = calculate_stock_vals(stock3,i,spend_on_each_stock)
    share_count_stock4, cost_of_buying_max_shares_stock4, cash_remain_stock4 = calculate_stock_vals(stock4,i,spend_on_each_stock)
    share_count_stock5, cost_of_buying_max_shares_stock5, cash_remain_stock5 = calculate_stock_vals(stock5,i,spend_on_each_stock)
    
    # total Cash remains for 5 stocks ['stock1','stock2','stock3','stock4','stock5']
    cash = cash_remain_stock1 + cash_remain_stock2 + cash_remain_stock3 + cash_remain_stock4 + cash_remain_stock5

    # total cost of buying max shares for 5 stocks ['stock1','stock2','stock3','stock4','stock5']
    cost = (cost_of_buying_max_shares_stock1 + cost_of_buying_max_shares_stock2 + cost_of_buying_max_shares_stock3
            + cost_of_buying_max_shares_stock4 + cost_of_buying_max_shares_stock5)

    # calculating MTM for i
    initial_MTM = cash + cost
    
    # calculation of weekly portfolio of 5 shares
    stocks = [stock1.split("_")[0],stock2.split("_")[0],stock3.split("_")[0],stock4.split("_")[0],stock5.split("_")[0]]
    shares_count = [share_count_stock1, share_count_stock2, share_count_stock3,share_count_stock4,share_count_stock5]
    shares_cost = [cost_of_buying_max_shares_stock1, cost_of_buying_max_shares_stock2, 
                                  cost_of_buying_max_shares_stock3,cost_of_buying_max_shares_stock4,
                                  cost_of_buying_max_shares_stock5]
    shares_cash = [cash_remain_stock1, cash_remain_stock2, cash_remain_stock3,cash_remain_stock4, cash_remain_stock5]
    data = {'max shares count':shares_count, 'max shares cost':shares_cost, 'remaining cash':shares_cash}
    global portfolio_df
    portfolio_df = pd.DataFrame(data, index=stocks)
    
    return initial_MTM

    new_cost_of_buying_max_shares_stock5 = share_count_stock5 * universe[stock5][i]

    MTM = int(new_cost_of_buying_max_shares_stock1 + new_cost_of_buying_max_shares_stock2
              + new_cost_of_buying_max_shares_stock3 + new_cost_of_buying_max_shares_stock4 + new_cost_of_buying_max_shares_stock5)
    return MTM

def calculate_Percentage_Change_AdjClose(i):
    # stocks % change in adj_close every 5 days
    # % change = (current_price - previous_price) / previous_price = (current_price / previous_price) - 1

    IBM_adj_close_change = ((universe['IBM_Adj Close'].loc[i] / universe['IBM_Adj Close'].loc[i-5]) - 1 )*100
    MSFT_adj_close_change = ((universe['MSFT_Adj Close'].loc[i] / universe['MSFT_Adj Close'].loc[i-5]) - 1 )*100
    GOOG_adj_close_change = ((universe['GOOG_Adj Close'].loc[i] / universe['GOOG_Adj Close'].loc[i-5]) - 1 )*100
    AAPL_adj_close_change = ((universe['AAPL_Adj Close'].loc[i] / universe['AAPL_Adj Close'].loc[i-5]) - 1 )*100
    AMZN_adj_close_change = ((universe['AMZN_Adj Close'].loc[i] / universe['AMZN_Adj Close'].loc[i-5]) - 1 )*100
    FB_adj_close_change = ((universe['FB_Adj Close'].loc[i] / universe['FB_Adj Close'].loc[i-5]) - 1 )*100
    NFLX_adj_close_change = ((universe['NFLX_Adj Close'].loc[i] / universe['NFLX_Adj Close'].loc[i-5]) - 1 )*100
    TSLA_adj_close_change = ((universe['TSLA_Adj Close'].loc[i] / universe['TSLA_Adj Close'].loc[i-5]) - 1 )*100
    ORCL_adj_close_change = ((universe['ORCL_Adj Close'].loc[i] / universe['ORCL_Adj Close'].loc[i-5]) - 1 )*100
    SAP_adj_close_change = ((universe['SAP_Adj Close'].loc[i] / universe['SAP_Adj Close'].loc[i-5]) - 1 )*100

    stocks_universe = ['IBM', 'MSFT', 'GOOG', 'AAPL', 'AMZN', 'FB', 'NFLX', 'TSLA', 'ORCL', 'SAP']

    adj_close_percentage_change = [IBM_adj_close_change,MSFT_adj_close_change,GOOG_adj_close_change,
                                   AAPL_adj_close_change,AMZN_adj_close_change,FB_adj_close_change,
                                   NFLX_adj_close_change,TSLA_adj_close_change,ORCL_adj_close_change,
                                   SAP_adj_close_change]

    data_adj_drop = {'stocks':stocks_universe,'adj_close_drop':adj_close_percentage_change}

    stocks_adj_close_price_change = pd.DataFrame(data_adj_drop)

    #choosing 5 stocks in 5 day period that dropped the most on 'Adj Close' prices
    stocks_adj_close_price_change.sort_values(by=['adj_close_drop'],inplace=True)
    
    stocks_adj_close_price_change['stocks'] = \
    stocks_adj_close_price_change['stocks'].apply(lambda x: "{}{}".format(x,'_Close'))
    
    five_stocks_buying_low = pd.DataFrame(stocks_adj_close_price_change.iloc[0:5]['stocks'])
    return five_stocks_buying_low
